# 项目文件夹结构

## Backend（Python + FastAPI）

```
backend/
├── config.yaml                             # 项目配置文件（相机/重建/检测/估计器）
├── pyproject.toml                          # Python 项目元数据
├── requirements.txt                        # Python 依赖清单
├── server/
│   ├── __init__.py
│   ├── main.py                             # FastAPI 应用入口、CORS、注册路由
│   ├── api/                                # Layer 3 — 后端API层
│   │   ├── __init__.py
│   │   ├── routes/
│   │   │   ├── __init__.py
│   │   │   ├── detection.py                # 检测结果查询接口
│   │   │   ├── reconstruction.py           # 重建触发/状态/结果接口
│   │   │   └── vehicle.py                  # 小车遥测 + 帧数据上报 + 位置查询
│   │   └── schemas/
│   │       ├── __init__.py
│   │       ├── detection.py                # 检测 Pydantic 模型
│   │       ├── reconstruction.py           # 重建 Pydantic 模型
│   │       └── vehicle.py                  # 遥测/帧/位置 Pydantic 模型
│   ├── engine/                             # Layer 4 — 重建引擎（单例）
│   │   ├── __init__.py
│   │   ├── engine.py                       # 帧累积管理 + 触发重建调度
│   │   └── frame_buffer.py                 # 帧缓冲区
│   ├── estimation/                         # Layer 4.5 — 位置估计器（单例）
│   │   ├── __init__.py
│   │   └── estimator.py                    # 自行车模型航位推算
│   ├── reconstruction/                     # 三维重建算法
│   │   ├── __init__.py
│   │   ├── mapper.py                       # 点云拼接/网格生成
│   │   └── fusion.py                       # 全量/增量融合
│   ├── detection/                          # YOLO 检测服务
│   │   ├── __init__.py
│   │   ├── detector.py                     # 模型加载与推理
│   │   └── postprocess.py                  # NMS、坐标映射后处理
│   ├── preprocessing/                      # Layer 3.5 — 点云转换层
│   │   ├── __init__.py
│   │   └── alignment.py                    # 深度图 → 点云（针孔模型）
│   ├── config/
│   │   └── __init__.py                     # YAML 配置加载器
│   └── utils/
│       ├── __init__.py
│       └── timestamp.py                    # 时间戳工具
└── tests/
    ├── __init__.py
    ├── test_api.py
    ├── test_engine.py
    └── test_estimator.py
```

## Frontend（Vue 3 + Vite + Electron）

```
frontend/
├── index.html                              # Vite 入口 HTML
├── package.json                            # 项目依赖与脚本
├── tsconfig.json                           # TypeScript 配置
├── vite.config.ts                          # Vite 构建配置
├── electron-builder.json5                  # Electron 打包配置
├── electron/                               # Electron 主进程
│   ├── main.ts                             # 主进程入口，创建窗口
│   └── preload.ts                          # 预加载脚本，暴露 IPC 桥接
├── src/
│   ├── App.vue                             # 根组件
│   ├── main.ts                             # Vue 应用入口
│   ├── config/
│   │   └── defaults.ts                     # 默认服务器IP、端口、轮询间隔
│   ├── api/                                # Layer 2 — 前端API层
│   │   ├── index.ts                        # API 统一入口
│   │   ├── detection.ts                    # 检测 API
│   │   ├── reconstruction.ts               # 重建 API
│   │   └── vehicle.ts                      # 小车数据 API
│   ├── network/                            # Layer 0 — 网络接发层
│   │   ├── http-client.ts                  # HTTP 请求封装
│   │   └── websocket-client.ts             # WebSocket 连接管理
│   ├── services/                           # 业务服务层
│   │   ├── data-loader/                    # Layer 0 — 本地文件读取
│   │   │   └── local-loader.ts
│   │   ├── data-saver/                     # Layer 0.5 — 数据保存
│   │   │   └── data-saver.ts
│   │   ├── pack-unpack/                    # Layer 1 — 包装与解包
│   │   │   └── codec.ts
│   │   ├── renderer/                       # Layer 1 — 渲染引擎
│   │   │   ├── three-scene.ts              # Three.js 场景管理
│   │   │   ├── point-cloud.ts              # 点云渲染
│   │   │   └── annotations.ts             # 检测框标注渲染
│   │   └── preprocessing/                  # Layer 1.5 — 前端预处理
│   │       ├── pointcloud.ts               # 深度图预处理
│   │       └── image.ts                    # 图像预处理
│   ├── composables/                        # Vue 组合式函数
│   │   ├── useConnection.ts                # 连接状态
│   │   ├── usePolling.ts                   # 轮询逻辑
│   │   └── useReconstruction.ts            # 重建流程
│   ├── components/                         # 共享 UI 组件
│   │   ├── layout/
│   │   │   ├── AppSidebar.vue              # 左侧导航栏
│   │   │   └── AppStatusBar.vue            # 顶部状态栏
│   │   ├── common/
│   │   │   ├── FileDropZone.vue            # 拖拽上传
│   │   │   └── ConfigPanel.vue             # 配置面板
│   │   └── three/
│   │       ├── SceneView.vue               # 三维场景画布
│   │       └── PointCloudView.vue          # 点云可视化
│   ├── views/                              # 页面视图（6 个界面）
│   │   ├── MainView.vue                    # 界面一：主界面
│   │   ├── RealtimeModeling.vue            # 界面二：实时建模
│   │   ├── ReplayModeling.vue              # 界面三：回放建模
│   │   ├── DefectDetail.vue                # 界面四：缺陷详情
│   │   ├── StaticDetection.vue             # 界面五：静态检测
│   │   └── SettingsView.vue                # 界面六：配置
│   ├── router/
│   │   └── index.ts                        # Vue Router 路由
│   ├── stores/                             # Pinia 状态管理
│   │   ├── connection.ts                   # 连接状态
│   │   ├── vehicle.ts                      # 小车数据
│   │   ├── reconstruction.ts               # 重建状态
│   │   └── settings.ts                     # 配置状态
│   ├── types/                              # TypeScript 类型
│   │   ├── api.ts                          # API 类型
│   │   ├── data.ts                         # 业务数据类型
│   │   └── settings.ts                     # 配置类型
│   └── utils/
│       ├── format.ts                       # 格式化工具
│       └── math.ts                         # 数学工具
└── public/                                 # 静态资源
```
